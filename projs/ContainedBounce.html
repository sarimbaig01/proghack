#include <SFML/Graphics.hpp>
#include <SFML/System.hpp>

int main() {
    //Create a graphics window, put a green ball at the top left of it
    sf::RenderWindow window(sf::VideoMode(800, 600), "Contained Bounce");
    sf::CircleShape ball(10);
    ball.setFillColor(sf::Color::Green);

    //Show the initial setup
    window.draw(ball);
    window.display();

    //Continue while the window is still open (hasn't been closed)        
    while (window.isOpen()) {
        sf::Event event;
        //While there are still events to process in the events queue     
        while (window.pollEvent(event)) {
            //Currently, the only event being processed is window closure
            if (event.type == sf::Event::Closed)
                window.close();
            }
            ///other events may be be tested in else-if statements
        }
    }
    return 0;
}
